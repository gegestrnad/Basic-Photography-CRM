   230→      window.handleLocalImport = function(e) {
   231→        var file = e.target.files[0];
   232→        if (!file) return;
   233→        if (!confirm('Import will REPLACE all current data. Continue?')) {
   234→          e.target.value = '';
   235→          return;
   236→        }
   237→        var reader = new FileReader();
   238→        reader.onload = function(ev) {
   239→          try {
   240→            DataStore.importData(ev.target.result);
   241→            showToast('Data imported successfully');
   242→            setTimeout(function() { location.reload(); }, 800);
   243→          } catch(err) {
   244→            showToast('Import failed: ' + err.message);
   245→          }
   246→        };
   247→        reader.readAsText(file);
   248→        e.target.value = '';
   249→      };

   251→      // Cloud Backup via Web Share API (works in WebView/APK)
   252→      async function shareBackupToCloud() {
   253→        var data = DataStore.exportData();
   254→        var blob = new Blob([data], { type: 'application/json' });
   255→        var fileName = 'photography-backup-' + new Date().toISOString().split('T')[0] + '.json';
   256→        var file = new File([blob], fileName, { type: 'application/json' });
   257→
   258→        if (navigator.canShare && navigator.canShare({ files: [file] })) {
   259→          try {
   260→            await navigator.share({
   261→              title: 'Photography Tracker Backup',
   262→              text: 'Backup exported on ' + new Date().toLocaleDateString(),
   263→              files: [file]
   264→            });
   265→            showToast('✅ Shared — pick Google Drive, OneDrive, etc.');
   266→          } catch(err) {
   267→            if (err.name !== 'AbortError') {
   268→              console.error('Share failed:', err);
   269→              showToast('Share failed: ' + err.message);
   270→            }
   271→          }
   272→        } else {
   273→          // Fallback: download to Downloads folder
   274→          var url = URL.createObjectURL(blob);
   275→          var a = document.createElement('a');
   276→          a.href = url;
   277→          a.download = fileName;
   278→          a.click();
   279→          URL.revokeObjectURL(url);
   280→          showToast('⚠️ Share not supported — saved to Downloads instead');
   281→        }
   282→      }

   305→      // Excel / CSV Export
   306→      function exportToExcel() {
   307→        try {
   308→          var jobs = DataStore.getJobs();
   309→          var pays = DataStore.getPayments();
   310→          var tasks = DataStore.getTasks();
   311→          var lists = DataStore.getLists();
   312→
   313→          var wb = XLSX.utils.book_new();
   314→
   315→          // Jobs sheet
   316→          var jobsData = jobs.map(function(j) {
   317→            return {
   318→              'Job ID': j.id,
   319→              'Client': j.clientName,
   320→              'Event Date': j.eventDate ? excelDate(j.eventDate) : '',
   321→              'Event Type': j.eventType,
   322→              'Location': j.location,
   323→              'Status': j.status,
   324→              'Package': j.packageName,
   325→              'Package Price': j.packagePrice,
   326→              'Add-ons Total': j.addonsTotal || 0,
   327→              'Total Price': (j.packagePrice || 0) + (j.addonsTotal || 0),
   328→              'Deposit Paid': j.depositPaid || 0,
   329→              'Balance Due': ((j.packagePrice || 0) + (j.addonsTotal || 0)) - (j.depositPaid || 0),
   330→              'Created': j.createdAt ? excelDate(j.createdAt) : '',
   331→              'Notes': j.notes || ''
   332→            };
   333→          });
   334→          var wsJobs = XLSX.utils.json_to_sheet(jobsData);
   335→          XLSX.utils.book_append_sheet(wb, wsJobs, 'Jobs');
   336→
   337→          // Payments sheet
   338→          var paysData = pays.map(function(p) {
   339→            var job = jobs.find(function(j) { return j.id === p.jobId; });
   340→            return {
   341→              'Payment ID': p.id,
   342→              'Job ID': p.jobId,
   343→              'Client': job ? job.clientName : '',
   344→              'Date': p.date ? excelDate(p.date) : '',
   345→              'Amount': p.amount,
   346→              'Method': p.method,
   347→              'Notes': p.notes || ''
   348→            };
   349→          });
   350→          var wsPays = XLSX.utils.json_to_sheet(paysData);
   351→          XLSX.utils.book_append_sheet(wb, wsPays, 'Payments');
   352→
   353→          // Tasks sheet
   354→          var tasksData = tasks.map(function(t) {
   355→            var job = jobs.find(function(j) { return j.id === t.jobId; });
   356→            return {
   357→              'Task ID': t.id,
   358→              'Job ID': t.jobId,
   359→              'Client': job ? job.clientName : '',
   360→              'Title': t.title,
   361→              'Status': t.done ? 'Done' : 'Pending',
   362→              'Due Date': t.dueDate ? excelDate(t.dueDate) : '',
   363→              'Created': t.createdAt ? excelDate(t.createdAt) : ''
   364→            };
   365→          });
   366→          var wsTasks = XLSX.utils.json_to_sheet(tasksData);
   367→          XLSX.utils.book_append_sheet(wb, wsTasks, 'Tasks');
   368→
   369→          // Lists sheet (reference data)
   370→          var listsData = {
   371→            'Event Types': lists.eventTypes || [],
   372→            'Package Names': lists.packageNames || [],
   373→            'Payment Methods': lists.paymentMethods || [],
   374→            'Task Templates': (lists.taskTemplates || []).map(function(t) { return t.title; }),
   375→            'Statuses': lists.statuses || []
   376→          };
   377→          var listsRows = [];
   378→          Object.keys(listsData).forEach(function(key) {
   379→            listsData[key].forEach(function(val, i) {
   380→              if (!listsRows[i]) listsRows[i] = {};
   381→              listsRows[i][key] = val;
   382→            });
   383→          });
   384→          var wsLists = XLSX.utils.json_to_sheet(listsRows);
   385→          XLSX.utils.book_append_sheet(wb, wsLists, 'Reference Lists');
   386→
   387→          // Wage Distributions sheet
   388→          var wds = DataStore.getWageDistributions();
   389→          if (wds.length > 0) {
   390→            var wdData = [];
   391→            wds.forEach(function(w) {
   392→              var job = jobs.find(function(j) { return j.id === w.jobId; });
   393→              if (w.breakdown && w.breakdown.length > 0) {
   394→                w.breakdown.forEach(function(b) {
   395→                  wdData.push({
   396→                    'WD ID': w.id,
   397→                    'Job ID': w.jobId,
   398→                    'Client': job ? job.clientName : '',
   399→                    'Date': w.created ? excelDate(w.created) : '',
   400→                    'Total Paid': w.totalPaid || 0,
   401→                    'Dist. Base': w.distributableBase || 0,
   402→                    'Staff': b.staffName || '',
   403→                    'Roles': (b.roles || []).join(', '),
   404→                    'Percentage': Math.round((b.percentage || 0) * 100) + '%',
   405→                    'Amount': b.amount || 0
   406→                  });
   407→                });
   408→              } else {
   409→                wdData.push({
   410→                  'WD ID': w.id,
   411→                  'Job ID': w.jobId,
   412→                  'Client': job ? job.clientName : '',
   413→                  'Date': w.created ? excelDate(w.created) : '',
   414→                  'Total Paid': w.totalPaid || 0,
   415→                  'Dist. Base': w.distributableBase || 0,
   416→                  'Staff': '', 'Roles': '', 'Percentage': '', 'Amount': ''
   417→                });
   418→              }
   419→            });
   420→            var wsWds = XLSX.utils.json_to_sheet(wdData);
   421→            XLSX.utils.book_append_sheet(wb, wsWds, 'Wage Distributions');
   422→          }
   423→
   424→          // Set column widths
   425→          ['Jobs', 'Payments', 'Tasks', 'Reference Lists', 'Wage Distributions'].forEach(function(name) {
   426→            var ws = wb.Sheets[name];
   427→            if (ws) {
   428→              var range = XLSX.utils.decode_range(ws['!ref'] || 'A1');
   429→              var cols = [];
   430→              for (var c = range.s.c; c <= range.e.c; c++) {
   431→                var cellAddr = XLSX.utils.encode_cell({ r: 0, c: c });
   432→                var cell = ws[cellAddr];
   433→                var width = cell && cell.v ? String(cell.v).length + 4 : 12;
   434→                cols.push({ wch: Math.min(Math.max(width, 10), 40) });
   435→              }
   436→              ws['!cols'] = cols;
   437→            }
   438→          });
   439→
   440→          var fileName = 'photography-tracker-' + new Date().toISOString().split('T')[0] + '.xlsx';
   441→          XLSX.writeFile(wb, fileName);
   442→          showToast('✅ Excel exported: ' + fileName);
   443→        } catch(err) {
   444→          console.error('Excel export error:', err);
   445→          showToast('Export failed: ' + err.message);
   446→        }
   447→      }

   699→      window.saveWageSettings = function() {
   700→        // Rules
   701→        var ruleNameInputs = document.querySelectorAll('.rule-name');
   702→        var rulePctInputs = document.querySelectorAll('.rule-pct');
   703→        var rules = [];
   704→        var valid = true;
   705→
   706→        for (var i = 0; i < ruleNameInputs.length; i++) {
   707→          var name = ruleNameInputs[i].value.trim();
   708→          var pct = parseFloat(rulePctInputs[i].value);
   709→          if (!name || isNaN(pct) || pct <= 0) continue;
   710→          rules.push({ role: name, percentage: pct / 100 });
   711→        }
   712→
   713→        var totalPct = rules.reduce(function(sum, r) { return sum + Math.round(r.percentage * 100); }, 0);
   714→        if (totalPct !== 100) {
   715→          if (!confirm('Role percentages total ' + totalPct + '%, not 100%. Save anyway?')) return;
   716→        }
   717→
   718→        DataStore.updateWageRules(rules);
   719→
   720→        // Ratio
   721→        var ratio = parseFloat(document.getElementById('wageRatioInput').value);
   722→        if (!isNaN(ratio) && ratio > 0) {
   723→          DataStore.updateWageConfig({ distributableRatio: ratio });
   724→        }
   725→
   726→        // Staff roles
   727→        var staff = DataStore.getLists().staff || [];
   728→        var checkboxes = document.querySelectorAll('.staff-role-cb');
   729→        var staffRoleMap = {};
   730→
   731→        checkboxes.forEach(function(cb) {
   732→          var name = cb.getAttribute('data-staff');
   733→          var role = cb.getAttribute('data-role');
   734→          if (!staffRoleMap[name]) staffRoleMap[name] = [];
   735→          if (cb.checked) staffRoleMap[name].push(role);
   736→        });
   737→
   738→        staff.forEach(function(s) {
   739→          if (staffRoleMap[s.name]) {
   740→            DataStore.updateStaffRoles(s.name, staffRoleMap[s.name]);
   741→          }
   742→        });
   743→
   744→        showToast('✅ Wage settings saved');
   745→        renderWageSettings();
   746→      };
